18   def geraGraphico(_col_pontos, _col_campanha, _parametro,  _idleg, _idclasse): 
19    
20       ID_LEGISLACAO = _idleg 
21       ID_CLASSE = _idclasse 
22    
23       cores = [ '#73A374', '#b0ccb0', '#a8cdd7', '#c0beaf', '#cec597', '#e8b7b7', '#66A7B9', '#6a6755', 
24                   '#B4E789', '#F4AABA', '#94AECD', '#C2C174', '#8B878F', '#C3A3D7', 
25                   '#77B25A', '#E7ACBC', '#7E93A6', '#93B483','#BC8A97','#B6C6D6', 
26                   '#D6C7A1','#9FA0A1','#4E763F','#495D71','#9F6270','#725E29','#5E5C57', 
27                   '#B4E789', '#F4AABA', '#94AECD', '#C2C174', '#8B878F', '#C3A3D7', 
28                   '#77B25A', '#E7ACBC', '#7E93A6', '#93B483','#BC8A97','#B6C6D6', 
29                   '#D6C7A1','#9FA0A1','#4E763F','#495D71','#9F6270','#725E29','#5E5C57', 
30               ] 
31    
32       template = """{chart: {width:800, plotBackgroundColor: '#E6E6E6'  }, 
33   title: { text: ' ' }, 
34   xAxis: [{ categories: {{categories|safe}} , crosshair: true, labels: {style: {fontWeight: 'bold'}}}], 
35   yAxis: [{title:{ text: '{{ str_param }}',style: {fontWeight: 'bold'} }, labels: { style: {fontWeight: 'bold'}, format: '{value:.{{casas}}f}' }}], 
36   credits: { text: ' ', href: ' '}, 
37   plotOptions: { line: { lineWidth: 2, marker: { enabled: false}}}, 
38   series: [ {% for item in series %} 
39   { color:'{{item.color}}', borderColor:'#E6E6E6', data:{{item.data}}, name:'{{ item.name}}', type:'{{item.type}}'}, 
40   {% endfor %}]} 
41   """ 
42    
43       ptomonit = [] 
44       campanha = [] 
45       grade = {} 
46    
47    
48       """ 
49       Seleciona no banco os registros a serem tratados 
50       """ 
51       col0 = Medicao.\ 
52             objects.\ 
53             filter( Parametro_FK_id=_parametro).\ 
54             values("Parametro_FK_id", 
55                    "Campanha_FK_id", 
56                    "PtoMonit_FK_id", 
57                    "vlrLbl", 
58                    "vlr").\ 
59             order_by("Parametro_FK_id", 
60                      "Campanha_FK_id", 
61                      "PtoMonit_FK_id") 
62    
63       col = [] 
64       for item in col0: 
65           if item["PtoMonit_FK_id"] in _col_pontos and \ 
66              item["Campanha_FK_id"] in _col_campanha: 
67    
68               if '<' not in item['vlrLbl']: 
69                   col.append(item) 
70    
71    
72       """ 
73       Cria os pontos e as campanhas 
74       """ 
75       for item in col: 
76           key = item["PtoMonit_FK_id"] 
77           if not key in ptomonit: 
78               ptomonit.append(key) 
79    
80           key = item["Campanha_FK_id"] 
81           if not key in campanha: 
82               campanha.append(key) 
83    
84       """ 
85       Cria grade Campanhas X Pontos 
86       """ 
87       for item in col: 
88           camp = item["Campanha_FK_id"] 
89           if camp in grade: 
90               linha = grade[camp] 
91           else: 
92               linha = [0 for itpto in ptomonit] 
93    
94           pto = item["PtoMonit_FK_id"] 
95           index = ptomonit.index(pto) 
96           linha[index] = item["vlr"] 
97    
98           grade[camp] = linha 
99    
100   
101      lim_min = {} 
102      lim_max = {} 
103      """ Determina Limites """ 
104      col_limite  = Limites.objects.filter(parametro_FK_id = _parametro, 
105                                       legislacao_FK_id = ID_LEGISLACAO, 
106                                       classe_FK_id = ID_CLASSE) 
107      if col_limite: 
108          obj_limite = col_limite[0] 
109          if obj_limite.vlr_max != 0: 
110              linha = [ float(obj_limite.vlr_max) for it in ptomonit ] 
111              lim_max = { 'name':'VMP', 'type':'line', 'color':'red', 'data':linha } 
112   
113          if obj_limite.vlr_min != 0: 
114              color = 'blue' if lim_max  else 'red' 
115              linha = [ float(obj_limite.vlr_min) for it in ptomonit ] 
116              lim_min = { 'name':'VMP', 'type':'line', 'color':color, 'data':linha } 
117   
118      """ 
119      Cria cria categorias 
120      """ 
121      categories = [] 
122      for item in ptomonit: 
123          ponto = PtoMonit.objects.get(pk=item).sigla 
124          categories.append('{0}'.format(ponto)) 
125   
126   
127   
128      """ 
129      Cria  as series 
130      """ 
131      series = [] 
132      pos = 0 
133      for item in grade: 
134          obj_campanha = Campanha.objects.get(pk=item).nome 
135          dados = [ float(it) for it in grade[item] ] 
136          if sum(dados) != 0: 
137              registro = {'name': '{0}'.format(obj_campanha), 
138                          'type':'column', 
139                          'data':dados, 
140                          'color': cores[pos]} 
141              series.append(registro) 
142              pos += 1 
143   
144      if len(series) == 0: 
145          return None 
146   
147      """ 
148      Adiciona limites se houver 
149      """ 
150      if lim_max: 
151          series.append(lim_max) 
152   
153      if lim_min: 
154          series.append(lim_min) 
155   
156   
157      """ 
158      Texto contendo o parâmetro do gráfico 
159      """ 
160      parametro = Param.objects.get(pk=_parametro) 
161      str_param = u'{0} ({1})'.format(parametro.nome , parametro.unidade_FK) 
162   
163      """ 
164      Renderiza o template 
165      """ 
166      contexto = Context({    'categories' : categories, 
167                              'str_param': str_param, 
168                              'casas': parametro.decimais, 
169                              'series':series}) 
170   
171      templ = Template(template) 
172      saida = templ.render(contexto) 
173      saida= '{"infile":"' + saida.encode('utf-8','ignore') + '","constr":"Chart"}' 
174      saida = saida.replace('\n','') 
175   
176      return saida

180  def geraTabela(document, _col_pontos, _col_campanha, _idleg, _idclasse): 
181   
182   
183      query = Medicao.objects.values('Parametro_FK_id', 'PtoMonit_FK_id', 'Campanha_FK_id', 'id') 
184   
185      col = [] 
186      qs = [] 
187      for item in query: 
188          if item["PtoMonit_FK_id"] in _col_pontos and \ 
189             item["Campanha_FK_id"] in _col_campanha: 
190   
191              reg = (item['Parametro_FK_id'], item['PtoMonit_FK_id']) 
192              if reg not in col: 
193                  col.append(reg) 
194                  qs.append(item) 
195   
196      df = pd.DataFrame.from_records(qs) 
197      saida = pd.pivot_table(df, index=[ 'Parametro_FK_id'], columns=['PtoMonit_FK_id'], values=['id'], fill_value=0) 
198   
199      db_param  =  list(Param.objects.all().values('id', 'nome', 'unidade_FK__sigla', 'decimais').order_by('nome')) 
200      db_pontos  = list(PtoMonit.objects.all().values('id', 'sigla').order_by('sigla')) 
201   
202      colunas = [ item[1] for item in  list(saida.columns)] 
203      pontos = [ item for item in db_pontos if item['id'] in colunas ] 
204      param = [ item for item in db_param if item['id'] in list(saida.index)] 
205   
206      linhas = [] 
207   
208      """Header""" 
209      linhaBranco = [ '' for item in colunas] 
210   
211      for x in param: 
212          linhas.append(list(linhaBranco)) 
213   
214      for key_param in saida.index: 
215          reg_param = [ item for item in param if item['id'] == key_param][0] 
216          pos_param = param.index(reg_param) 
217   
218          for key_pto in saida.loc[key_param].index: 
219              reg_pto = [ item for item in pontos if item['id'] == key_pto[1] ][0] 
220              pos_pto = pontos.index(reg_pto) 
221   
222              key_vlr = saida.loc[key_param].get(key_pto) 
223              if key_vlr == 0: 
224                  vlr = '' 
225              else: 
226                  if key_param == 840: 
227                      vlr = Medicao.objects.get(pk=key_vlr).data 
228                  else: 
229                      record = Medicao.objects.get(pk=key_vlr) 
230                      simbol = '' 
231                      if '<' in record.vlrLbl: 
232                          simbol = '<' 
233                      if '>' in record.vlrLbl: 
234                          simbol = '>' 
235   
236                      casas = param[pos_param]['decimais'] 
237                      vlr = '{0}{1:.{2}f}'.format(simbol, record.vlr, casas).replace('.',',') 
238   
239              linhas[pos_param][pos_pto] = vlr 
240   
241   
242      pos = 0 
243      col_linhas = [] 
244      saida = [] 
245      for item in linhas: 
246          key = param[pos]['id'] 
247          insere = True 
248          if param[pos]['id'] == 840: 
249              newlinha = [ '{0:02d}/{1:02d}/{2}'.format(it.day, it.month, it.year) for it in item] 
250              newlinha.insert(0, ('Data', '', '') ) 
251              col_linhas.append(newlinha) 
252   
253              newlinha = ['{0:02d}:{1:02d}'.format(it.hour, it.minute) for it in item] 
254              newlinha.insert(0, ('Hora', '', '') ) 
255              col_linhas.append(newlinha) 
256              insere = False 
257   
258   
259          if param[pos]['id'] == 798: 
260              newlinha =list(item) 
261              newlinha.insert(0, ('Profundidade', 'm', '') ) 
262              col_linhas.append(newlinha) 
263              insere = False 
264   
265          if insere: 
266              vleg = int(_idleg) 
267              vclass = int(_idclasse) 
268              vparam = param[pos]['id'] 
269              col_limite  = Limites.objects.filter(parametro_FK_id =vparam, 
270                                                   legislacao_FK_id = vleg, 
271                                                   classe_FK_id = vclass) 
272   
273              if len(col_limite): 
274                  casas = param[pos]['decimais'] 
275                  vlr_max = col_limite[0].vlr_max 
276                  if  vlr_max == 0: 
277                      vlr_max = '' 
278                  else: 
279                      vlr_max = '{0:.{1}f}'.format(vlr_max, casas).replace('.',',') 
280   
281                  vlr_min = col_limite[0].vlr_min 
282                  if  vlr_min == 0: 
283                      vlr_min = '' 
284                  else: 
285                      vlr_min = '{0:.{1}f}'.format(vlr_min, casas).replace('.',',') 
286   
287   
288                  if vlr_max  and vlr_min: 
289                      vlr = vlr_min + ' - ' + vlr_max 
290                  elif vlr_min: 
291                      vlr = vlr_min 
292                  elif vlr_max: 
293                      vlr = vlr_max 
294   
295   
296   
297              else: 
298                  vlr = '' 
299   
300              reg = [ (it['nome'], it['unidade_FK__sigla'], vlr) for it in param if it['id']==key][0] 
301              item.insert(0, reg) 
302              saida.append(item) 
303          pos +=1 
304   
305      for it in col_linhas: 
306         saida.insert(0,it) 
307   
308      main_rel = [ item[1:] for item in saida ] 
309      left_rel = [ item[:1][0] for item in saida ] 
310      head_rel = [ item['sigla'] for item in pontos] 
311   
312      step_col = 7 
313      start = 0 
314      total = len(main_rel[1]) 
315   
316      for start in range(0,total,step_col): 
317          document.add_page_break() 
318          final = min( start+step_col , total) 
319          table = document.add_table(rows=len(saida)+1, cols=(final-start)+3) 
320          table.autofit = True 
321          table.style= 'Table Grid' 
322   
323          hdr_cells = table.rows[0].cells 
324          hdr_cells[0].text = 'Ponto' 
325          hdr_cells[1].text = 'Unidade' 
326          hdr_cells[2].text = 'VMP' 
327          pos = 3 
328          for ptos in head_rel[start:final]: 
329              hdr_cells[pos].text = ptos 
330              pos += 1 
331   
332          row = 0 
333          for item in main_rel: 
334             pos = 0 
335             row_cells = table.rows[row+1].cells 
336             for it in left_rel[row]: 
337                 row_cells[pos].text = u'{0}'.format(it) 
338                 pos += 1 
339   
340             for coluna in item[start:final]: 
341                 row_cells[pos].text = u'{0}'.format(coluna) 
342                 pos += 1 
343             row += 1 
344   
345          for row in range(len(saida)+1): 
346              for col in range( (final-start)+3): 
347                      table.cell(row,col).paragraphs[0].paragraph_format.alignment=WD_ALIGN_PARAGRAPH.CENTER 
348   
349      document.add_page_break()

352  def processa(_col_pontos, _col_campanha, _col_parametro,  _idleg, _idclasse): 
353   
354      document = Document('/home/wbeirigo/Clima/weather/proj/matriz.docx') 
355   
356      URL_PHANTOM = 'http://10.3.0.29:3003' 
357      headers ={ 'Content-Type': 'application/json', '-X POST':'' } 
358      indice = 0 
359   
360   
361      section = document.add_section() 
362      section.orientation = WD_ORIENT.LANDSCAPE 
363      indice += 1 
364      document.add_paragraph('  ', style='Normal') 
365      document.add_paragraph('  ', style='Normal') 
366      document.add_heading(u'Quadro 6.{0} - - Resultados do monitoramento da qualidade da água'.format(indice, ''), level=2) 
367      geraTabela(document, _col_pontos, _col_campanha, _idleg, _idclasse) 
368      document.add_paragraph('  ', style='Normal') 
369      section = document.add_section() 
370      section.orientation = WD_ORIENT.PORTRAIT 
371      for parametro  in _col_parametro: 
372   
373          """ Não processa parametros Data e Profundidade""" 
374          if parametro in (798,840): 
375              continue 
376   
377          saida = geraGraphico(_col_pontos, _col_campanha, parametro, _idleg, _idclasse) 
378          if not saida: 
379              continue 
380          r = requests.post(URL_PHANTOM, data=saida, headers=headers) 
381          png_recovered = base64.decodestring(r.content) 
382          path = "/tmp/{0}.png".format(int(time.time()*1000)) 
383          f = open(path, "w") 
384          f.write(png_recovered) 
385          f.close() 
386   
387          indice += 1 
388          obj_param = Param.objects.get(pk=parametro) 
389          nome = obj_param.nome 
390          texto = obj_param.texto 
391          document.add_heading(u'6.{0} - {1}'.format(indice, nome), level=3) 
392          document.add_paragraph('  ', style='Normal') 
393          document.add_paragraph(texto, style='Normal') 
394          document.add_paragraph('  ', style='Normal') 
395          document.add_paragraph('  ', style='Normal') 
396          document.add_paragraph(u'Figura 6.{0} -Resultados obtidos para o parâmetro {1}'.format(indice, nome), style='f') 
397          document.add_picture(path, width=Inches(5.2)) 
398          document.add_paragraph('  ', style='Normal') 
399          last_paragraph = document.paragraphs[-1] 
400          last_paragraph.alignment = WD_ALIGN_PARAGRAPH.CENTER 
401          document.add_page_break() 
402          os.remove(path) 
403      caminho = '/media/projeto_cemig/01_SOFTWARE/rel/' 
404      path = caminho + "{0}.docx".format(int(time.time()*1000)) 
405      document.save(path) 
406      return path
